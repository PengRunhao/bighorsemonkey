//统计球鞋总数

{

 //按球鞋价格排序排序
void sumpaixu(struct Shoes* head)
{
	int choice,count=0,i,j;
	struct Shoes *p1=head,*p2,*temp,*t;
	printf("\n\t\t\t\t\t            按照球鞋价格排序        \n ");
	printf("\n\t\t\t\t\t|                                       |");
	printf("\n\t\t\t\t\t|           1.降序                      |");
	printf("\n\t\t\t\t\t|                                       |");
	printf("\n\t\t\t\t\t|           2.升序                      |");
	printf("\n\t\t\t\t\t|                                       |");
	printf("\n\t\t\t\t\t|           3.返回上一层                |"); 
	printf("\n\t\t\t\t\t|                                       |");
	printf("\n\t\t\t\t\t|           4.退出                      |");
	printf("\n\t\t\t\t\t|_______________________________________|");
	printf("\n\t\t\t\t\t请选择:");
	scanf("%d",&choice);
	while(p1=p1->next)
		count++;
	switch(choice)
	{
		case 1:
			{
				for(i=1;i<count;i++)
				{
					p1=head->next;
					p2=p1->next;
					temp=head;
					for(j=0;j<count-1;j++)
					{
						if(p1->price  > p2->price) 
						{
							p1->next=p2->next;
							p2->next=p1;
							temp->next=p2;
							t=p2;
							p2=p1;
							p1=t;
							
						}
						p1=p1->next;
						p2=p2->next;
						temp=temp->next;
					}
				}
				printf("\n\n\t\t\t\t\t降序排列为:");
				print(head);
			}
			break;
		case 2:
			{
				for(i=1;i<count;i++)
				{
					p1=head->next;
					p2=p1->next;
					temp=head;
					for(j=0;j<count-1;j++)
					{
						if(p1->price <p2->price)
						{
							p1->next=p2->next;
							p2->next=p1;
							temp->next=p2;
							t=p2;
							p2=p1;
							p1=t;
							
						}
						p1=p1->next;
						p2=p2->next;
						temp=temp->next;
					}	
				}
				printf("\n\n\t\t\t\t\t升序排列为:");
				print(head);		
			}
			break;
		case 3:
			system("cls");
			fflush(stdin);
			printf("\n\n\t\t\t\t\t按任意键返回上一层\n");
			getch(); 
			system("cls");	
			sort();
				break;
		default:
			break;
	}
}
//根据球鞋名称及球鞋序号混合查找

{

	printf("\n\n\t\t\t\t\t请输入球鞋序号");
