#include <stdio.h>

int main(void)
{
	int n;
	scanf("%d", &n);
	int i, j, k = 1;  //k为赋值计数器；
	int x, y, dir;   // x,y为行列坐标，dir为 方向变量；
	int arr[10][10];
	for (i = 1; i <= 2*n-1; i++) {
		if (i % 2 == 1) {
			x = i - 1;
			y = 0;
			dir = -1;
		}
		else {
			x = 0;
			y = i - 1;
			dir = 1;
		}
		while (x >= 0 && y >= 0) {
			if (x < n && y < n) {
				arr[x][y] = k;
				k++;
			}
			x += dir;
			y -= dir;
		}
	}
	for (i = 0; i < n; i++) {
		for (j = 0; j < n; j++) {
			if (j != 0) printf(" ");
			printf("%d", arr[i][j]);
		}
		printf("\n");
	}
	 return 0;
}
